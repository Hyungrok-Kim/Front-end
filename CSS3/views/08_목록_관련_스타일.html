순서가 있는 목록 스타일

        선택자 {
            list-style-type : 
                    decimal // 기본값
                    decimal-leading-zero // 0부터 시작하는 순서
                    lower-alpha // 영 소문자 
                    upper-alpha // 영 대문자
                    lower-roman // 로마 소문자
                    upper-roman // 로마 대문자
        }
